def estrategia_mmovel(serie,longmean=100,shortmean=10):
    #Recebe Uma srie de preos de Fechamento
    serie =pd.DataFrame(serie)

    


    #Marca o cruzamento da mdia longa com a mdia Curta
    serie['roll']   =  serie.rolling(shortmean,min_periods=5).mean()-serie.rolling(longmean,min_periods=5).mean()
    

    serie['DECISAO']=np.nan
    #0 significa compra 1 Significa Fazer nada, 0 significa compra pois  util em uma funo futura,venda poderia ser -1
    serie.loc[serie['roll']>0.0,'DECISAO']=0
    serie.loc[serie['roll']<=0.0,'DECISAO'] =1

def calculate_returns(x):
    
    x['DAILY_CHANGE']=np.nan
    #Simples Variao Diria   
    x['DAILY_CHANGE']= x.iloc[:,0]/x.iloc[:,0].shift(1)
    #Seleciona as trocas de  posies 
    trocas=x['ESTRAT']==0
    
    #Seciona todas as posies
    comprado=x['DECISAO']==0
    
    
    #x.loc[comprado,'TOTAL_CHANGE']=np.cumprod(x.loc[comprado,'DAILY_CHANGE'].values)
    #Prenche a variao Diria Somente para aqueles onde temos posio
    x.loc[:,'DAILY_CHANGE']=x.loc[comprado,'DAILY_CHANGE']
    #Onde se trocou de posio ainda no h rendimento  ento 0% de variao
    x.loc[trocas,'DAILY_CHANGE']=1
    
    return x

def stop_loss(x,stop):
    # um gato, mas funciona
    #Soma acumulada da deciso == 0 quando se esta comprado,ento autoamticamente cada compra ganhya um codigo
    x['SUM']=x['DECISAO'].cumsum()
    #Calcula o resultado dentro de cada operao
    x["OP_RESULT"]=x[x['DECISAO']==0].groupby(['SUM'])['DAILY_CHANGE'].cumprod()
    #filtro para operaes abaixo do stop
    f=x['OP_RESULT']<stop
    #pega a primeira data onde aquela operao entrou em stop   
    f=x[f]['SUM'].reset_index().groupby(['SUM']).min()
    #cria um filtro para todas as datas subsequentes dentro da mesma operao
    f=pd.DataFrame(index=x.SUM).join(f)
    #filtra as operaes que entrata em stop
    x=x[~(x.index>=f['Date']) &( f.index==x.SUM)]
    return x
